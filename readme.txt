SELECT
    MIN(BALANCE_DATE) AS min_balance_date,
    MAX(BALANCE_DATE) AS max_balance_date,
    COUNT(DISTINCT BALANCE_DATE) AS distinct_balance_dates
FROM SonataODS.dbo.ACCOUNT_GENERAL_BALANCE;


SELECT
    CAST(BALANCE_DATE AS date) AS balance_date,
    COUNT(*) AS row_count,
    COUNT(DISTINCT ACCOUNT_ID) AS account_count
FROM SonataODS.dbo.ACCOUNT_GENERAL_BALANCE
WHERE BALANCE_DATE >= '2024-12-01'
  AND BALANCE_DATE <  '2025-04-15'
GROUP BY CAST(BALANCE_DATE AS date)
ORDER BY balance_date;
