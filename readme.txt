SELECT
    TRANSACTION_TYPE_CODE,
    TRANSACTION_TYPE_SHORT_DESC,
    COUNT(*) AS row_count,
    MIN(GROSS_AMOUNT) AS min_amount,
    MAX(GROSS_AMOUNT) AS max_amount
FROM SonataODS.dbo.ACCOUNT_TRANSACTION
WHERE
       UPPER(TRANSACTION_TYPE_SHORT_DESC) LIKE '%ROLLOVER%'
    OR UPPER(TRANSACTION_TYPE_SHORT_DESC) LIKE '%TRANSFER%'
    OR UPPER(TRANSACTION_TYPE_SHORT_DESC) LIKE '%WITHDRAW%'
    OR UPPER(TRANSACTION_TYPE_SHORT_DESC) LIKE '%SWITCH OUT%'
GROUP BY
    TRANSACTION_TYPE_CODE,
    TRANSACTION_TYPE_SHORT_DESC
ORDER BY row_count DESC;
